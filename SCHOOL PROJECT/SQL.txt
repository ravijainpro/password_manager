CREATE DATABASE PROJECT;
USE PROJECT;
CREATE TABLE user (USER VARCHAR(256), PASS VARCHAR(256), EMAIL VARCHAR(256), DATE INT(2), DAY VARCHAR(3), YEAR INT(4), GENDER VARCHAR(10));
CREATE TABLE details (NAME VARCHAR(256),EMAIL VARCHAR(256),PHONE INT(10),PASSWORD VARCHAR(256),USERNAME VARCHAR(256),PASSKEY VARCHAR(256));
CREATE TABLE login (ID INT(255) NOT NULL AUTO_INCREMENT, USERNAME VARCHAR(256),PASSWORD VARCHAR(256),PRIMARY KEY(ID));
